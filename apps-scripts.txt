// SETUP (one-time):
//   1. Extensions → Apps Script → Project Settings → Script Properties
//   2. Add: WEBHOOK_URL  = https://ninja-database.vercel.app
//           WEBHOOK_SECRET = (any secret string — must match Vercel env var SHEETS_WEBHOOK_SECRET)
// ─────────────────────────────────────────────────────────────────────────────

// ── Webhook helper ────────────────────────────────────────────────────────────
function callWebhook(payload) {
  var props = PropertiesService.getScriptProperties();
  var url    = props.getProperty('WEBHOOK_URL');
  var secret = props.getProperty('WEBHOOK_SECRET');
  if (!url || !secret) return; // silently skip if not configured
  payload.secret = secret;
  try {
    UrlFetchApp.fetch(url + '/api/sheets-webhook', {
      method: 'POST',
      contentType: 'application/json',
      payload: JSON.stringify(payload),
      muteHttpExceptions: true
    });
  } catch(e) {
    console.error('Webhook failed:', e.message);
  }
}

function onOpen(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const buttonSheet = ss.getSheetByName('Buttons');
  const dataSheet = ss.getSheetByName('data');
  var nameRange = dataSheet.getRange(2, 1, dataSheet.getLastRow());
  var buttonCell = buttonSheet.getRange(1, 2);
  var rule = SpreadsheetApp.newDataValidation().requireValueInRange(nameRange, true).build();
  buttonCell.setDataValidation(rule);
  const updateSheet = ss.getSheetByName('Update Progress');
  const progressSheet = ss.getSheetByName('Progress');
  var progressNameRange = progressSheet.getRange(2, 1, progressSheet.getLastRow());
  var selectCell = updateSheet.getRange(1, 2);
  var rule2 = SpreadsheetApp.newDataValidation().requireValueInRange(progressNameRange, true).build();
  selectCell.setDataValidation(rule2);
}

function onEdit(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const updateSheet = ss.getSheetByName('Update Progress');
  const progressSheet = ss.getSheetByName('Progress');
  const logProgressSheet = ss.getSheetByName('Progress Log');
  const range = e.range;
  const sheet = range.getSheet();
  const row = range.rowStart;
  const col = range.columnStart;
  const wStart = 3;
  const yStart = 5;
  const oStart = 8;
  const gStart = 11;
  const blStart = 14;
  const pStart = 17;
  const brStart = 38;
  const rStart = 70;
  const blackStart = 74;

  if(sheet.getName() === 'data' && col === 4) {
    if(e.value === "TRUE") {
      var addedCell = sheet.getRange(row, 6);
      var added = addedCell.getValue();
      var numCell = sheet.getRange(row, 3);
      var num = numCell.getValue();
      if (typeof num == 'number') {
        addedCell.setValue(added += num);
        numCell.clear();
      } else {
        numCell.setValue('NaN');
      }
      range.uncheck();
    }
  }
  else if(sheet.getName() === 'data' && col === 5) {
    if(e.value === "TRUE") {
      var addedCell = sheet.getRange(row, 6);
      var added = addedCell.getValue();
      var numCell = sheet.getRange(row, 3);
      var num = numCell.getValue();
      if (typeof num == 'number') {
        addedCell.setValue(added -= num);
        numCell.clear();
      } else {
        numCell.setValue('NaN');
      }
      range.uncheck();
    }
  }

  if(sheet.getName() === 'Progress' && col === 2) {
    var checkRange = null;
    var clearRange = progressSheet.getRange(row, 3, 1, progressSheet.getLastColumn());
    clearRange.uncheck();
    switch(e.value) {
      case 'White':   break;
      case 'Yellow':  var checkRange = progressSheet.getRange(row, 3, 1, 2);   break;
      case 'Orange':  var checkRange = progressSheet.getRange(row, 3, 1, 5);   break;
      case 'Green':   var checkRange = progressSheet.getRange(row, 3, 1, 8);   break;
      case 'Blue':    var checkRange = progressSheet.getRange(row, 3, 1, 11);  break;
      case 'Purple':  var checkRange = progressSheet.getRange(row, 3, 1, 14);  break;
      case 'Brown':   var checkRange = progressSheet.getRange(row, 3, 1, 35);  break;
      case 'Red':     var checkRange = progressSheet.getRange(row, 3, 1, 67);  break;
      case 'Black':   var checkRange = progressSheet.getRange(row, 3, 1, 71);  break;
      default: break;
    }
    if(checkRange != null) checkRange.check();
  }

  if(sheet.getName() === 'Update Progress' && row === 2 && col > 2) {
    var selectName = updateSheet.getRange(1, 2);
    var dataStart = progressSheet.getRange(1, 1);
    var name = selectName.getValue();
    var textFinder = ss.createTextFinder(name).matchCase(true).startFrom(dataStart);
    var foundName = textFinder.findNext();
    var nameRow = foundName.getRow();
    var selectBelt = progressSheet.getRange(nameRow, 2);
    var belt = selectBelt.getValue();
    var changeCell = null;
    var nextBelt = null;
    var isUnity = false;
    switch(belt) {
      case 'White':   changeCell = progressSheet.getRange(nameRow, (col + wStart) - 3);   nextBelt = 'Yellow';  break;
      case 'Yellow':  changeCell = progressSheet.getRange(nameRow, (col + yStart) - 3);   nextBelt = 'Orange';  break;
      case 'Orange':  changeCell = progressSheet.getRange(nameRow, (col + oStart) - 3);   nextBelt = 'Green';   break;
      case 'Green':   changeCell = progressSheet.getRange(nameRow, (col + gStart) - 3);   nextBelt = 'Blue';    break;
      case 'Blue':    changeCell = progressSheet.getRange(nameRow, (col + blStart) - 3);  nextBelt = 'Purple';  break;
      case 'Purple':  changeCell = progressSheet.getRange(nameRow, (col + pStart) - 3);   nextBelt = 'Brown';   isUnity = true; break;
      case 'Brown':   changeCell = progressSheet.getRange(nameRow, (col + brStart) - 3);  nextBelt = 'Red';     isUnity = true; break;
      case 'Red':     changeCell = progressSheet.getRange(nameRow, (col + rStart) - 3);   nextBelt = 'Black';   isUnity = true; break;
      case 'Black':   changeCell = progressSheet.getRange(nameRow, (col + blackStart) - 3); isUnity = true;     break;
      default: break;
    }

    if(changeCell != null) {
      if(e.value === "TRUE") {
        changeCell.check();
        var now = new Date();
        var dateString = Utilities.formatDate(now, 'America/Regina', 'MMMM dd, yyyy');
        var activityRange = updateSheet.getRange(1, col);
        var activityName = activityRange.getValue();
        var namedRanges = ss.getNamedRanges();
        for(var foo in namedRanges) {
          if(namedRanges[foo].getName() == 'Unity' && isUnity) {
            var unityNamedRange = namedRanges[foo].getRange();
            var newRow = unityNamedRange.getLastRow() + 1;
            namedRanges[foo].setRange(logProgressSheet.getRange('G3:K' + newRow));
          } else if(namedRanges[foo].getName() == 'Degrees' && !isUnity) {
            var degreeNamedRange = namedRanges[foo].getRange();
            var newRow = degreeNamedRange.getLastRow() + 1;
            namedRanges[foo].setRange(logProgressSheet.getRange('A3:E' + newRow));
          }
        }
        if(isUnity) {
          var unityRange = logProgressSheet.getRange(unityNamedRange.getLastRow() + 1, 7, 1, 4);
          var unityCheck = logProgressSheet.getRange(unityRange.getRow(), 11);
          if(range.getColumn() == updateSheet.getLastColumn()) {
            unityRange.setValues([[dateString, name, belt + " -> " + nextBelt, activityName]]);
            unityCheck.insertCheckboxes();
            selectBelt.setValue(nextBelt);
            loadData();
          } else {
            unityRange.setValues([[dateString, name, belt, activityName]]);
            unityCheck.insertCheckboxes();
          }
        } else {
          var degreeRange = logProgressSheet.getRange(degreeNamedRange.getLastRow() + 1, 1, 1, 4);
          var degreeCheck = logProgressSheet.getRange(degreeRange.getRow(), 5);
          if(range.getColumn() == updateSheet.getLastColumn()) {
            degreeRange.setValues([[dateString, name, belt + " -> " + nextBelt, activityName]]);
            degreeCheck.insertCheckboxes();
            selectBelt.setValue(nextBelt);
            loadData();
          } else {
            degreeRange.setValues([[dateString, name, belt, activityName]]);
            degreeCheck.insertCheckboxes();
          }
        }
      }
      else if(e.value === "FALSE") {
        changeCell.uncheck();
        var namedRanges = ss.getNamedRanges();
        for(var foo in namedRanges) {
          if(namedRanges[foo].getName() == 'Unity' && isUnity) {
            var unityNamedRange = namedRanges[foo].getRange();
          } else if(namedRanges[foo].getName() == 'Degrees' && !isUnity) {
            var degreeNamedRange = namedRanges[foo].getRange();
          }
        }
        var activityRange = updateSheet.getRange(1, col);
        var activityName = activityRange.getValue();
        var logProgressStart = logProgressSheet.getRange(1, 1);
        var textFinder = ss.createTextFinder(name).matchCase(true).startFrom(logProgressStart);
        var foundNameRanges = textFinder.findAll();
        for(var i of foundNameRanges) {
          if(i.getSheet().getName() == 'Progress Log' && logProgressSheet.getRange(i.getRow(), i.getColumn() + 1).getValue() == belt && logProgressSheet.getRange(i.getRow(), i.getColumn() + 2).getValue() == activityName) {
            if(isUnity) {
              logProgressSheet.getRange(i.getRow(), i.getColumn() - 1, 1, 5).clear();
              logProgressSheet.getRange(i.getRow() + 1, i.getColumn() - 1, unityNamedRange.getLastRow(), 5).moveTo(logProgressSheet.getRange(i.getRow(), i.getColumn() - 1));
            } else {
              logProgressSheet.getRange(i.getRow(), i.getColumn() - 1, 1, 5).clear();
              logProgressSheet.getRange(i.getRow() + 1, i.getColumn() - 1, degreeNamedRange.getLastRow(), 5).moveTo(logProgressSheet.getRange(i.getRow(), i.getColumn() - 1));
            }
          }
        }
      }
    }
  }
}

function loadData() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const updateSheet = ss.getSheetByName('Update Progress');
  const progressSheet = ss.getSheetByName('Progress');
  var clearHeaderRange = updateSheet.getRange(1, 3, 1, updateSheet.getLastColumn());
  clearHeaderRange.clear();
  var clearRange = updateSheet.getRange(2, 2, 1, updateSheet.getLastColumn());
  clearRange.removeCheckboxes();
  clearRange.clear();
  var selectName = updateSheet.getRange(1, 2);
  var dataStart = progressSheet.getRange(1, 1);
  var name = selectName.getValue();
  var textFinder = ss.createTextFinder(name).matchCase(true).startFrom(dataStart);
  var foundName = textFinder.findNext();
  var nameRow = foundName.getRow();
  var selectBelt = progressSheet.getRange(nameRow, 2);
  var belt = selectBelt.getValue();
  var rangeColor = "#ffffff";
  switch(belt) {
    case 'White':   var headerRange = progressSheet.getRange(1, 3, 1, 2);   var ninjaRange = progressSheet.getRange(nameRow, 3, 1, 2);   break;
    case 'Yellow':  var headerRange = progressSheet.getRange(1, 5, 1, 3);   var ninjaRange = progressSheet.getRange(nameRow, 5, 1, 3);   rangeColor = "#fff2cc"; break;
    case 'Orange':  var headerRange = progressSheet.getRange(1, 8, 1, 3);   var ninjaRange = progressSheet.getRange(nameRow, 8, 1, 3);   rangeColor = "#fce5cd"; break;
    case 'Green':   var headerRange = progressSheet.getRange(1, 11, 1, 3);  var ninjaRange = progressSheet.getRange(nameRow, 11, 1, 3);  rangeColor = "#d9ead3"; break;
    case 'Blue':    var headerRange = progressSheet.getRange(1, 14, 1, 3);  var ninjaRange = progressSheet.getRange(nameRow, 14, 1, 3);  rangeColor = "#cfe2f3"; break;
    case 'Purple':  var headerRange = progressSheet.getRange(1, 17, 1, 21); var ninjaRange = progressSheet.getRange(nameRow, 17, 1, 21); rangeColor = "#d9d2e9"; break;
    case 'Brown':   var headerRange = progressSheet.getRange(1, 38, 1, 32); var ninjaRange = progressSheet.getRange(nameRow, 38, 1, 32); rangeColor = "#c5b190"; break;
    case 'Red':     var headerRange = progressSheet.getRange(1, 70, 1, 4);  var ninjaRange = progressSheet.getRange(nameRow, 70, 1, 4);  rangeColor = "#f4cccc"; break;
    case 'Black':   var headerRange = progressSheet.getRange(1, 74, 1, 1);  var ninjaRange = progressSheet.getRange(nameRow, 74, 1, 1);  rangeColor = "#838383"; break;
  }
  var headerData = headerRange.getValues();
  var showData = ninjaRange.getValues();
  var beltRange = updateSheet.getRange(2,2);
  var headerOutRange = updateSheet.getRange(1, 3, 1, headerData[0].length);
  var outRange = updateSheet.getRange(2, 3, 1, showData[0].length);
  headerOutRange.setBackground(rangeColor);
  outRange.setBackground(rangeColor);
  beltRange.setBackground(rangeColor);
  beltRange.setValue(belt);
  headerOutRange.setValues(headerData);
  outRange.insertCheckboxes();
  outRange.setValues(showData);
}

function checkLeaderboard(name) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const oldNBLog = ss.getSheetByName('Old NB Logs');
  const logSheet = ss.getSheetByName('NB Log');
  var now = new Date();
  var dateString = Utilities.formatDate(now, 'America/Regina', 'MMMM');

  var lbRange = logSheet.getRange(2, 1, 5, 2);
  var lb = lbRange.getValues();
  var lastMonth = logSheet.getRange(1, 2).getValue();
  if (dateString != lastMonth) {
    var monthlyLeaderRange = oldNBLog.getRange(1, oldNBLog.getLastColumn() + 1, 4, 1);
    monthlyLeaderRange.setValues([[lastMonth], [lb[0][0] + ' (' + lb[0][1] + ')'], [lb[1][0] + ' (' + lb[1][1] + ')'], [lb[2][0] + ' (' + lb[2][1] + ')']]);
    lbRange.setValues([['Ninja', 0], ['Ninja', 0], ['Ninja', 0], ['Ninja', 0], ['Ninja', 0]]);
    logSheet.getRange(1, 2).setValue(dateString);
    var oldLogs = logSheet.getRange(9, 1, logSheet.getLastRow() - 9, 4);
    var oldVals = oldLogs.getValues();
    var movePosition = oldNBLog.getRange(oldNBLog.getLastRow() + 1, 1, logSheet.getLastRow() - 9, 4);
    movePosition.setValues(oldVals);
    logSheet.deleteRows(9, logSheet.getLastRow() - 9);
  }
  var namedRanges = ss.getNamedRanges();
  for(var foo in namedRanges) {
    if(namedRanges[foo].getName() == 'Leaderboard') {
      var rangeToSort = namedRanges[foo].getRange();
    }
  }
  var exists = false;
  var pos = 0;
  for(var elem of rangeToSort.getValues()) {
    if(elem[0] === name){ exists = true; break; }
    pos++;
  }
  var ninjaBucksTotal = 0;
  var logStart = logSheet.getRange(1, 1);
  var logTextFinder = ss.createTextFinder(name).matchCase(true).startFrom(logStart);
  var logFoundNames = logTextFinder.findAll();
  for(var i of logFoundNames) {
    if(i.getSheet().getName() == 'NB Log' && i.getRow() > 8) {
      var ninjaBucksValue = parseInt(logSheet.getRange(i.getRow(), i.getColumn() + 2).getValue().toString().replace("+", ""));
      ninjaBucksTotal += ninjaBucksValue;
    }
  }
  if(exists) { lb[pos][1] = ninjaBucksTotal; lbRange.setValues(lb); }
  else if (ninjaBucksTotal > lb[4][1]) { lb[4][0] = name; lb[4][1] = ninjaBucksTotal; lbRange.setValues(lb); }
  rangeToSort.sort({column: 2, ascending: false});
}

function removeLogRow() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const dataSheet = ss.getSheetByName('data');
  const logSheet = ss.getSheetByName('NB Log');
  var delRowCell = logSheet.getRange(1, 4);
  var delRow = delRowCell.getValue();
  if (typeof delRow == 'number') {
    var rowToDelete = logSheet.getRange(delRow, 1, 1, 4);
    var rowInfo = rowToDelete.getValues();
    var name = rowInfo[0][1];
    var ninjaBucksValue = parseInt(rowInfo[0][3].toString().replace("+", ""));
    var dataStart = dataSheet.getRange(1, 1);
    var textFinder = ss.createTextFinder(name).matchCase(true).startFrom(dataStart);
    var foundName = textFinder.findNext();
    var nameRow = foundName.getRow();
    var addedCell = dataSheet.getRange(nameRow, 6);
    var added = addedCell.getValue();
    addedCell.setValue(added -= ninjaBucksValue);
    logSheet.deleteRow(delRow);
    checkLeaderboard(name);
    // Sync removal to website
    callWebhook({ type: 'nb_award', ninjaName: name, amount: -ninjaBucksValue, reason: 'Log Row Removed' });
    delRowCell.setValue('');
  } else {
    delRowCell.setValue('NaN');
  }
}

// postToDiscord now also syncs belt changes to the website
function postToDiscord() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const logProgressSheet = ss.getSheetByName('Progress Log');
  var discordUrl = 'https://discord.com/api/webhooks/1434259585492979905/--yxO4mgcyZT_OeXwsZyLTopDBuPVL6NseMFhZPWMn3aT-Ns0cAToSFgbhn4FwrcqDgA';
  var namedRanges = ss.getNamedRanges();
  for(var foo in namedRanges) {
    if(namedRanges[foo].getName() == 'Unity') {
      var unityNamedRange = namedRanges[foo].getRange();
    } else if(namedRanges[foo].getName() == 'Degrees') {
      var degreeNamedRange = namedRanges[foo].getRange();
    }
  }
  for(var i = 0; i < unityNamedRange.getNumRows(); i++) {
    var info = logProgressSheet.getRange(i + 3, 7, 1, 5).getValues();
    if(info[0][4] == false) {
      if(info[0][2].toString().includes('-')) {
        var message = info[0][1] + ' to ' + info[0][2].toString().split(' ')[2] + ' Belt!';
        var payload = JSON.stringify({content: message});
        var params = { headers: {'Content-Type': 'application/json'}, method: "POST", payload: payload, muteHttpExceptions: true };
        UrlFetchApp.fetch(discordUrl, params);
        logProgressSheet.getRange(i + 3, 11).check();
        updateTracking(info[0][1], info[0][2].toString().split(' ')[0], info[0][3], true, true);
        updateDashboard(info[0][1]);
        // Sync belt change to website
        var parts = info[0][2].toString().split(' -> ');
        callWebhook({ type: 'belt_update', ninjaName: info[0][1], newBelt: parts[1] || parts[0], notes: info[0][3] });
      } else {
        logProgressSheet.getRange(i + 3, 11).check();
      }
    }
  }
  for(var i = 0; i < degreeNamedRange.getNumRows(); i++) {
    var info = logProgressSheet.getRange(i + 3, 1, 1, 5).getValues();
    if(info[0][4] == false) {
      if(info[0][2].toString().includes('-')) {
        var message = info[0][1] + ' to ' + info[0][2].toString().split(' ')[2] + ' Belt!';
        var payload = JSON.stringify({content: message});
        var params = { headers: {'Content-Type': 'application/json'}, method: "POST", payload: payload, muteHttpExceptions: true };
        UrlFetchApp.fetch(discordUrl, params);
        logProgressSheet.getRange(i + 3, 5).check();
        updateTracking(info[0][1], info[0][2].toString().split(' ')[0], info[0][3], true, false);
        updateDashboard(info[0][1]);
        // Sync belt change to website
        var parts = info[0][2].toString().split(' -> ');
        callWebhook({ type: 'belt_update', ninjaName: info[0][1], newBelt: parts[1] || parts[0], notes: info[0][3] });
      } else if (!info[0][2].toString().includes('-')) {
        var message = info[0][1] + ' finished ' + info[0][3] + ' ' + info[0][2] + ' Belt!';
        var payload = JSON.stringify({content: message});
        var params = { headers: {'Content-Type': 'application/json'}, method: "POST", payload: payload, muteHttpExceptions: true };
        UrlFetchApp.fetch(discordUrl, params);
        logProgressSheet.getRange(i + 3, 5).check();
        updateTracking(info[0][1], info[0][2], info[0][3], false, false);
        updateDashboard(info[0][1]);
      } else {
        logProgressSheet.getRange(i + 3, 5).check();
      }
    }
  }
}

function updateTracking(name, belt, activity, isBeltUp, isUnity) {
  const trackSS = SpreadsheetApp.openById("1zEb_QyZ78Zz1I08Btrfb4egcicqqKxLClrUwue506B4");
  const trackSheet = trackSS.getSheetByName("Tracking");
  trackSheet.insertRowBefore(2);
  const now = new Date();
  const dateString = Utilities.formatDate(now, 'America/Regina', 'MM/dd/yyyy');
  if(isUnity) {
    var outStr = belt + " Belt";
  } else {
    if(isBeltUp) { var outStr = belt + " Belt " + activity + "/Completion"; }
    else { var outStr = belt + " Belt " + activity; }
  }
  trackSheet.getRange(2, 1, 1, 3).setValues([[name, dateString, outStr]]);
}

function updateDashboard(name) {
  const dashSS = SpreadsheetApp.openById("1ybeUvRZFf-HtU_SNFgq65xb7Mc_OBBrtgDF9PI2mTlE");
  const dashSheet = dashSS.getSheetByName("Belts Totem");
  var dashStart = dashSheet.getRange(3, 1);
  var textFinder = dashSS.createTextFinder(name).matchCase(true).startFrom(dashStart);
  var foundName = textFinder.findNext();
  var col = foundName.getColumn();
  dashSheet.getRange(foundName.getRow(), foundName.getColumn()).clearContent();
  var newPos = dashSheet.getRange(dashSheet.getLastRow() + 1, foundName.getColumn() + 1);
  newPos.setValue(name);
  var firstCol = dashSheet.getRange(3, col, dashSheet.getLastRow());
  var secondCol = dashSheet.getRange(3, col + 1, dashSheet.getLastRow());
  firstCol.sort(col);
  secondCol.sort(col + 1);
}

function attack() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const bossSheet = ss.getSheetByName('Boss Battle');
  var damageRange = bossSheet.getRange(1, 2);
  var damage = damageRange.getValue();
  var bossHealthRange = bossSheet.getRange(1, 6);
  var bossHealth = bossHealthRange.getValue();
  if (typeof damage == 'number') {
    bossHealthRange.setValue(bossHealth - damage);
    damageRange.clear();
  } else {
    damageRange.clear();
  }
}
